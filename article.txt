function handleFullWidthBanner(selector) {
    if (selector === '#po-overlay-articlegate-full-width-full-article-scroll-below') {
        document.querySelector(selector).classList.add(selector.replace('#', ''));
    }
    document.querySelector(selector).style.minHeight = "254px";
    document.querySelector('#subscription-content').style.display = 'none';
    document.querySelectorAll('.send-us-your-news-cta').forEach((el) => { el.style.display = 'none'; });
    document.querySelectorAll('.mega-news-block').forEach((el) => { el.style.display = 'none'; });

    const block = document.querySelector(selector);
    const containerBlock = block.parentElement;
    const containerBlockHeight = 254; // to prevent abrupt jumping

    if (!block) { return; }

    block.style.display = 'block';

    const blockRect = block.getBoundingClientRect();
    const startTop = blockRect.top + window.scrollY;
    const blockHeight = blockRect.height;
    const screenWidth = window.screen.width;

    let lastState = ''; // 'top', 'bottom', 'default'

    function getScrollbarWidth() {
        return window.innerWidth - document.documentElement.clientWidth;
    }

    function updatePosition() {
        const scrollY = window.scrollY;
        const viewportHeight = window.innerHeight;
        const bottomTrigger = startTop + blockHeight - viewportHeight;

        let newState = 'default';

        if (scrollY > startTop && screenWidth >= 768) {
            newState = 'top';
        } else if (scrollY < bottomTrigger && screenWidth >= 768) {
            newState = 'bottom';
        } else if (scrollY < bottomTrigger + 90 && screenWidth < 768) {
            newState = 'bottom-mobile';
        }

        const scrollbarWidth = getScrollbarWidth();

        if (newState !== lastState) {
            switch (newState) {
                case 'top':
                    block.style.position = 'fixed';
                    block.style.top = '0';
                    block.style.bottom = 'auto';
                    block.style.marginLeft = '0';
                    block.style.marginRight = '0';
                    containerBlock.style.minHeight = `${containerBlockHeight}px`;
                    break;

                case 'bottom':
                    block.style.position = 'fixed';
                    block.style.bottom = '0';
                    block.style.top = 'auto';
                    block.style.marginLeft = '0';
                    block.style.marginRight = '0';
                    block.style.width = `calc(100vw - ${scrollbarWidth}px)`;
                    containerBlock.style.minHeight = 'unset';
                    break;

                case 'bottom-mobile':
                    block.style.position = 'fixed';
                    block.style.bottom = '90px';
                    block.style.top = 'auto';
                    block.style.marginLeft = 'calc(-100vw + 100%)';
                    block.style.marginRight = 'calc(-100vw + 100%)';
                    containerBlock.style.minHeight = 'unset';
                    break;

                default:
                    block.style.position = '';
                    block.style.top = '';
                    block.style.bottom = '';
                    block.style.width = `calc(100vw - ${scrollbarWidth}px)`;
                    block.style.marginLeft = '-100vw';
                    block.style.marginRight = '-100vw';
                    containerBlock.style.minHeight = 'unset';
                    break;
            }

            lastState = newState;
        }

        requestAnimationFrame(updatePosition);
    }

    updatePosition();
    window.addEventListener('resize', updatePosition);
}

                // flip pbjs boolean to signal to pbjs the ad server has already been called
                pbjs.adserverCalled = true;

                //flip boolean for adserver request to avoid duplicate requests
                requestManager.sendAdserverRequest = true;

                // make ad request to DFP
                googletag.cmd.push(function() {
                    window.executionSet = performance.now();
                    apstag.setDisplayBids();
                    __iasPET.setTargetingForGPT();
                    // console.log("DFP: Refresh called, setting adslots to ", requestManager.adSlots);
                    if (requestManager.adSlots.length > 0){
                        googletag.pubads().refresh(requestManager.adSlots);
                    } else {
                        googletag.pubads().refresh();
                    }
                    setPerformanceCookie();
                });
            } else {
                window.setTimeout(function() {
                    sendAdserverRequest(true);
                }, 200);
            }
        }


        function requestBids(apstagSlots, bidTimeout) {
            // fetch apstag bids, set bid targting, then call headerBidderBack
            // to get the ads for the first time
            apstag.fetchBids({
                slots: apstagSlots,
                timeout: (bidTimeout - 200)
            }, function(bids) {
                headerBidderBack('a9');
            });


            var edktAudiences;

            // Here we fetch audiences which the user has been placed in.
            try {
                edktAudiences = JSON.parse(localStorage.getItem('edkt_matched_audience_ids') || '[]')
                    .slice(0, 100)
                    .map(String);
            } catch (e) {
                edktAudiences = [];
            }

            // request bids from prebid
            pbjs.que.push(function() {
                // console.log("DFP: PBJS request bids called");
                pbjs.rp.requestBids({
                    callback: headerBidderBack,
                    data: {
                        airgrid: {
                            perid: edktAudiences
                        },
                        ozoneSettings,
                        ozoneTargeting
                    },
                });
            });
        }

        //requestBids
        requestBids(apstagSlots, bidTimeout)

        // set timeout to send request to call sendAdserverRequest() after timeout
        // if all bidders haven't returned before then
        window.setTimeout(function() {
            sendAdserverRequest();
        }, bidTimeout);
    };

    function setPerformanceCookie() {
        var start = window.startExec;
        var prebidReady = window.prebidReady - start;
        var domComplete = window.domComplete - start;
        var googleComplete = window.googleComplete - start;
        var finished = window.executionSet - start;

        /*    var cVal = Math.round(prebidReady) + ':' +
                   Math.round(domComplete) + ':' +
                   Math.round(googleComplete) + ':' +
                   Math.round(finished); */
        var cVal = Math.round(finished);
        var t = new Date();
        t.setMilliseconds(t.getMilliseconds() + 600000);

        $.cookie('adtiming', cVal, {path: '/', domain: '.countypress.co.uk', expires: t});
    }

                <div id="po-inline-articlegate-fullarticle"><p class="article-first-paragraph">
  A project aiming to uncover the hidden histories of an Isle of Wight fort and its surrounding defences has received a major funding boost.
</p>
<div id="inArticleAd" class="advert-container mar-block-ad mar-block-ad--in-article"><div id="DFP_in_article_mpu"></div></div>
<div id="po-inline-articlewall-partialarticle-closable"></div>
<div id="po-inline-articlegate-partialarticle">
<div id="po-overlay-articlegate-full-width-scroll-below"></div>
<div id="subscription-content">
<p>
  The <a href="https://www.bembridgefort.org/?ref=ed_direct" target="_blank" rel="nofollow"><strong>Bembridge Fort Trust</strong></a> has been awarded £10,000 from the National Lottery Heritage Fund, working with the Hampshire and IW Community Foundation and Creative Island, under the Past Futures initiative.
</p>
<p>
  The grant, combined with a further £3,200 in donations, will fund ‘Forgotten People, Lost Technologies’, a project designed to bring together volunteers, schools, and community groups to explore the Fort’s rich and often secretive past.
</p>
<div id="inArticleAd-1" data-name-place="DFP_outbrain">
    <div class="OUTBRAIN" data-ob-contenturl="https://www.countypress.co.uk/news/25600081.hidden-wartime-stories-isle-wight-fort-uncovered/" data-ob-installation-key="NEWSQ15BEE3GLF6A6AL9EECN9" data-widget-id="AR_2"></div>
    <script>
        if (whiteList === true) {
            $('#inArticleAd-1').remove();
        }
    </script>
</div>
<p>
  The initiative builds on a series of successful heritage projects run by the Trust over the past year and will focus on the personal stories and technological innovations that shaped Bembridge Fort’s 150-year history.
</p>
<p>
  <img srcset="https://www.countypress.co.uk/resources/images/18956342/?type=mds-article-575 575w, https://www.countypress.co.uk/resources/images/18956342/?type=mds-article-962 962w, https://www.countypress.co.uk/resources/images/18956342/?type=mds-article-642 1400w, https://www.countypress.co.uk/resources/images/18956342/?type=mds-article-620 1401w" sizes="(max-width: 575px) 575px, (max-width: 992px) 962px, (max-width: 1400px) 1400px, (min-width: 1401px) 1401px" width="100%"><span class="inline-image-caption">Entrance of Bembridge Fort <em>(Image: IWCP)</em></span>
</p>
<p>
  The Trust is calling on Islanders whose ancestors served, worked, or lived near Bembridge Fort, or nearby military sites, to come forward with any photos, documents, or memories.
</p>
<div id="inArticleAd-2" class="advert-container mar-block-ad mar-block-ad--in-article"><div id="DFP_in_article_2"></div></div>
<p>
  Free workshops will be set up to record stories, and the team will show visitors how to restore, colourise, and even animate old photos.
</p>
<p>
  David Reeves, chair of the Bembridge Fort Trust, said the project was about ensuring the Island’s wartime history was not lost.
</p>
<p>
  “We look forward to recording and hearing your stories and make the past live again,” he said.
</p>
<div id="po-inline-article-inside" style="display: none; margin-bottom: 20px; clear: both;"></div>
<div id="inArticleAd-3" class="advert-container mar-block-ad mar-block-ad--in-article"><div id="DFP_teads"></div></div>
<p>
  “Any memories of your relative’s service matter, and we believe their stories need to be told. Let us make sure these stories are never forgotten.”
</p>
<hr>
<ul>
<li>
    <strong>Those wishing to contribute can contact the Trust by emailing</strong> <a href="mailto:info@bembridgefort.org" target="_blank"><strong>info@bembridgefort.org</strong></a><strong>.</strong>
  </li>
</ul>
<hr>
<p>
  <img srcset="https://www.countypress.co.uk/resources/images/20164485/?type=mds-article-575 575w, https://www.countypress.co.uk/resources/images/20164485/?type=mds-article-962 962w, https://www.countypress.co.uk/resources/images/20164485/?type=mds-article-642 1400w, https://www.countypress.co.uk/resources/images/20164485/?type=mds-article-620 1401w" sizes="(max-width: 575px) 575px, (max-width: 992px) 962px, (max-width: 1400px) 1400px, (min-width: 1401px) 1401px" width="100%"><span class="inline-image-caption">Bembridge Fort on Culver Down <em>(Image: Bembridge Fort Trust)</em></span>
</p>
<p>
  Trustee and project manager Tim Wander said the project would reveal the “untold personal stories that have been hidden behind the impressive walls, tunnels and dry ditches of Bembridge Fort.”
</p>
<div id="inArticleAd-4" class="advert-container mar-block-ad mar-block-ad--in-article"><div id="DFP_in_article_3"></div></div>
<p>
  “We hope to share the history of people and events in the context of the unique fort buildings and its ground-breaking technologies,” he said.
</p>
<p>
  “The stories may well include the continual threat of invasion, aerial attack, heartbreak, loss, and joy – all of wartime life carried on within the Fort’s huge walls as hundreds of now forgotten people were prepared to give everything for our freedom.”
</p>
<p>
  He added that researchers would also delve into wartime archives to uncover the defensive technologies developed in secret at the site during the Second World War.
</p>
<div id="inArticleAd-5" class="advert-container mar-block-ad mar-block-ad--in-article"><div id="DFP_outstream_2"></div></div>
<p>
  The Trust hopes Islanders will come forward with memories, photos, or documents connected to the Fort or nearby military sites.
</p>

function removeAdContainer(name) {
    const element = document.getElementById(name);
    // doesn't exist? exit
    if(!element) {
        return;
    }

    const parent = element.parentElement;
    if (parent && parent.classList.contains('block-bottom-margin')) {
        // if immediate parent contains block-bottom-margin as a class, remove the parent as well
        parent.classList.remove('block-bottom-margin');
    }
    if (parent && parent.classList.contains('mar-leaderboard--top')) {
        // in case of leaderboard, remove as has min-height
        parent.remove();
        return;

    if (parent && parent.classList.contains('mar-leaderboard--bottom')) {
        // in case of article ad container, remove as has min-height
        parent.remove();
        return;
    }

    element.remove();
}
